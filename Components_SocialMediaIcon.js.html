import React from 'react';

/**
 * Changes the links of the social media icon when the link in the input text box changes.
 * 
 * @param e
 */
export const onChangeLinkSocialMedia = (e) => {
	var link = document.getElementById(e.target.id);
	var sm_icon = document.getElementById('sm-icon-' + e.target.id);
	sm_icon.value = link.value;
	var sm_href = document.getElementById('href-sm-icon-' + e.target.id);
	sm_href.href = link.value;
}

/**
 * Toggles the display property of social media icons to allow the users the choice of their addition/removal.
 * 
 * @param e
 */
export const toggleCheckBoxSocialMedia = (e) => {
	var cbox = document.getElementById(e.target.id);
	var sm_icon = document.getElementById('sm-icon-sm-href_' + cbox.name + "_" + e.target.id.split("_")[2]);
	if (cbox.checked) {
		sm_icon.style.display = "block";
		var sm_href = document.getElementById('sm-href_' + cbox.name + "_" + e.target.id.split("_")[2])
		sm_href.style.display = "block";
	} else {
		sm_icon.style.display = "none";
		var sm_href = document.getElementById('sm-href_' + cbox.name + "_" + e.target.id.split("_")[2])
		sm_href.style.display = "none";
	}
}


/**

 * @class SocialMediaIcon

class SocialMediaIcon extends React.Component {

	constructor() {
		super(...arguments);
		this.ID = "sm-icon";
		this.type = "";
		this.icon = "";
		this.style = "";
	}

	/**
	 * Open the social media link in a new window
	 *
	 * @memberof SocialMediaIcon
	 */
	btnClick() {
		var x = document.getElementById(this.ID);
		// console.log(x);
		// console.log(x.value);
		window.open(x.value);
	}
