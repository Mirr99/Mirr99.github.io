import {
	HtmlEditor,
	Image,
	Inject,
	Link,
	QuickToolbar,
	RichTextEditorComponent,
	Toolbar,
	Resize
} from '@syncfusion/ej2-react-richtexteditor';
import React from 'react';

/**
 * Represents an editable text box. Provides capabilities like dynamic change in font, color etc.
 *
 * @export
 * @class EditableText
 * @extends {React.Component}
 */
export class EditableText extends React.Component {
	constructor(bgcolor) {
		super(...arguments);
		this.readonly = false;
		this.inlineMode = {
			enable: true,
			onSelection: true
		};
		this.toolbarSettings = {
			items: ['FontName', 'FontSize', 'FontColor',
				'Bold', 'Italic', 'Underline', 'Formats', '-',
				'Alignments', 'OrderedList', 'UnorderedList', 'CreateLink', 'BackgroundColor'
			]
		};
		this.fontFamily = {
			width: '100px'
		};
		this.fontSize = {
			width: '100px'
		};
		this.fontColor = {
			modeSwitcher: false
		};
	}
	/**
	 * Renders the editable text box element.
	 *
	 * @return {*} 
	 * @memberof EditableText
	 */
	render() {
		return (<div>
			<RichTextEditorComponent ref = {
				(scope) => { this.rteObj = scope; } } readonly = { this.readonly } height = { "auto" } inlineMode = { this
				.inlineMode } toolbarSettings = { this.toolbarSettings } fontColor = { this.fontColor } fontFamily = { this
				.fontFamily } fontSize = { this.fontSize }>
			<p class = 'para' > Text Placeholder </p>
			<Inject services = { [Toolbar, Image, Link, HtmlEditor, QuickToolbar] }/>
			</RichTextEditorComponent>
			</div>);
	}
}


/**
 * Represents a text box placeholder to ensure the user cannot interact with the text boxes in the style tab.  
 *
 * @export
 * @class EditableTextPlaceHolder
 * @extends {React.Component}
 */
export class EditableTextPlaceHolder extends React.Component {
	constructor(bgcolor) {
		super(...arguments);
		this.backgroundColor = bgcolor;
		this.readonly = true;
	}
	/**
	 * Renders the text box placeholder element.
	 *
	 * @return {*} 
	 * @memberof EditableTextPlaceHolder
	 */
	render() {
		return (<div>
			<RichTextEditorComponent ref = {
				(scope) => { this.rteObj = scope; } } readonly = { this.readonly } height = { "auto" }>
			<p class = 'para'> Text Placeholder </p>
			<Inject services = { [HtmlEditor] }/>
			</RichTextEditorComponent>
			</div>);
	}
}
