/**
 * Clones a given object.
 * 
 * @param obj Object to be cloned.
 */
export const clone = (obj) => {
	let copy

	// Handle the 4 simple types, and null or undefined
	if(obj === null || typeof obj !== 'object') {
		return obj
	}

	// Handle Date
	if(obj instanceof Date) {
		copy = new Date()
		copy.setTime(obj.getTime())
		return copy
	}

	// Handle Array
	if(obj instanceof Array) {
		copy = []
		for(let i = 0, len = obj.length; i < len; i++) {
			copy[i] = clone(obj[i])
		}
		return copy
	}

	// Handle Object
	if(obj instanceof Object) {
		copy = {}
		for(const attr in obj) {
			if(obj.hasOwnProperty(attr)) {
				copy[attr] = clone(obj[attr])
			}
		}
		return copy
	}

}

/**
 * Switches the panel between Style and Blocks.
 * 
 * @param idname Represents the panel ID.
 */
export const changePanel = (idName) => {
	if(document.getElementById(idName)) {
		var x = document.getElementsByClassName("panel");
		var i;
		for(i = 0; i < x.length; i++) {
			x[i].style.display = "none";
		}
		var y = document.getElementById(idName);
		y.style.display = "block";
	}
}

/**
 * Changes the corresponding block's background color when a color is picked.
 * 
 * @param e The selected div.
 */
export const changeDivColor = (e) => {
	var color_picker = document.getElementsByClassName("background_edit_color_picker")[0];
	var req_div_id = parseInt(color_picker.id.split("_")[4]);
	var req_div = document.getElementById(req_div_id);
	req_div.style.backgroundColor = e.currentValue.hex;
}

/**
 * Converts the given rgb values to the corresponding hex code.
 * 
 * @param rgb The rgb values.
 */
export const rgb2hex = (rgb) => {
	if(rgb.search("rgb") == -1) {
		console.log(rgb);
		return rgb;
	} else {
		rgb = rgb.match(/^rgba?\((\d+),\s*(\d+),\s*(\d+)(?:,\s*(\d+))?\)$/);

		function hex(x) {
			return ("0" + parseInt(x).toString(16)).slice(-2);
		}
		return "#" + hex(rgb[1]) + hex(rgb[2]) + hex(rgb[3]);
	}
}
