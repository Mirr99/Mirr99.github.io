/**
 * Changes the link of the form in the clicked div to the link mentioned in the style tab.
 *
 * @param e
 */
export const onChangeFormLink = (e) => {
	var link = document.getElementById(e.target.id);
	var form = document.getElementById('form_' + e.target.id);
	form.action = link.value;
}
