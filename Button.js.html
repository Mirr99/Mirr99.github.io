import {
	ButtonComponent
} from '@syncfusion/ej2-react-buttons';
import React from "react";

/**
 * Represents a button element
 * 
 * @class Button
 * @extends {React.Component}
 */
class Button extends React.Component {

		constructor() {
			super(...arguments);
			this.ID = "btn";
		}

		/**
		 * Opens the button's link in a new window
		 * 
		 * @memberof Button
		 */
		btnClick() {
			window.open(document.getElementById(this.ID).value);
		}
		/**
		 * Renders the button element while adding appropriate classes for styling
		 *
		 * @return {*} 
		 * @memberof Button
		 */
		render() {
				this.ID = this.props.id;

				return (<div>
			<ButtonComponent cssClass = 'e-link e-outline'
			value = ""
			id = {
				this.props.id
			}
			onClick = {
				this.btnClick.bind(this)
			}> Button </ButtonComponent>
			</div>);
	}
}

/**
 * Changes the link of the button in the clicked div to the link mentioned in the style tab.
 *
 * @param e
 */
export const onChangeLink = (e) => {
	var link = document.getElementById(e.target.id);
	var btn = document.getElementById('btn_' + e.target.id);
	btn.value = link.value;
}

export default Button;
